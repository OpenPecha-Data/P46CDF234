སྡོམ་པ་གསུམ་གྱི་སྨོན་ལམ་དོན་གཉིས་ངལ་བསོ་བཞུགས། །

ན་མོ། རྒྱལ་བ་ཀུན་གྱི་ཐུགས་བསྐྱེད་མཁའ་ཀློང་ལས། ། ཕན་བདེའི་སྤྲིན་གྱི་བླ་བྲེ་མཛེས་པའི་དབུས། ། སྡོམ་གསུམ་བདུད་རྩིའི་ཆར་རྒྱུན་འཇོ་བ་ཡིས། ། ཐར་པ་མཆོག་གི་མྱུ་གུ་བཞད་པར་ཤོག ༡ ། མཐོན་མཐིང་ལྕང་ལོའི་ཟུར་ཕུད་བྲེགས་པ་ཡིས། ། སྲིད་པའི་འཆིང་བ་མ་ལུས་བརྒལ་བྱས་ཏེ། ། ངུར་སྨྲིག་བླ་གོས་གསར་པའི་གཟི་འོད་ཀྱིས། ། ས་གསུམ་ཕན་བདེའི་སྣང་བས་ཁྱབ་གྱུར་ཅིག ༢ ། གཞི་གསུམ་ཆོ་གའི་རྩ་བ་ལེགས་བརྟན་ཅིང༌། ། སྤངས་དང་འཇུག་པའི་ཚུལ་ཁྲིམས་ཡལ་ག་རྒྱས། ། སྲིད་པའི་དགྲ་བཅོམ་གེ་སར་མཛེས་གྱུར་པའི། ། དམ་ཆོས་འདུལ་བའི་ཚལ་བརྒྱ་འཕེལ་བར་ཤོག ༣ ། གཞན་ཕན་ག་བུར་བརྩེགས་པའི་མཛའ་བཤེས་གྱིས། ། འགྲོ་ཀུན་དགའ་བའི་ཀུནྡ་རྒྱས་པ་དང༌། ། ལྷན་ཅིག་འཁོར་བའི་མུན་སྟུག་རབ་བཅོམ་སྟེ། ། ཟག་མེད་བདུད་རྩིའི་དཔལ་གྱིས་འཚོ་བར་ཤོག ༤ ། སྤྱིན་དང་ཚུལ་ཁྲིམས་བཟོད་པ་བརྩོན་འགྲུས་དང༌། ། བསམ་གཏན་ཤེས་རབ་ནོར་བུའི་འཁྲི་ཤིང་ཆེར། ། ཕ་རོལ་ཕྱིན་པའི་ལམ་ལ་བརྟེན་པ་ཡིས། ། བྱང་ཆུབ་མཆོག་གི་འབྱོར་པས་ཕྱུག་པར་ཤོག ༥ ། ལུང་དང་རྟོགས་པའི་ཆོས་ཚུལ་རྒྱ་མཚོ་ཆེ། ། ཤེས་རབ་གསུམ་གྱི་གྲུ་གཟིང་གཡོ་བ་ཡིས། ། རྒྱལ་གསུང་ལམ་བདེན་ཕ་མཐར་བརྒལ་བྱས་ནས། ། ཐམས་ཅད་མཁྱེན་པའི་གླིང་དུ་ཕྱིན་པར་ཤོག ༦ ། ལེགས་བཤད་ཉིན་བྱེད་གསར་པའི་འོད་ཟེར་གྱིས། ། སྐལ་བཟང་བློ་གྲོས་པདྨོ་བཞད་པ་དང༌། ། ལྷན་ཅིག་ལམ་མིན་མུན་པ་བསལ་བྱས་ཏེ། ། རྒྱལ་བསྟན་སྣང་བས་ཡོངས་སུ་ཁྱབ་པར་ཤོག ༧ ། ཆོས་བརྒྱད་འཆིང་བ་རབ་གྲོལ་དབེན་གནས་སུ། ། འདུ་འཛིའི་སྤྱོད་པ་མ་ལུས་རིང་དོར་ཏེ། ། རྩེ་གཅིག་ཏིང་འཛིན་མཆོག་ལ་གཡོ་མེད་པས། ། དོན་གཉིས་མ་ལུས་ལྷུན་གྱིས་གྲུབ་པར་ཤོག ༨ ། རྡོ་རྗེ་ཐེག་པའི་བརྒྱུད་འཛིན་བུམ་བཟང་ནས། ། སྨིན་བྱེད་བདུད་རྩིའི་ཐིགས་པས་བྲན་པ་ལས། ། ཐར་པའི་ཞིང་ས་རབ་ཏུ་བརླན་བྱས་ཏེ། ། གཡེལ་མེད་དམ་ཚིགས་བོན་ཆགས་པར་ཤོག ༩ ། སྐུ་གསུང་ཐུགས་ཀྱི་རྡོ་རྗེ་རབ་བཟུང་ཞིང༌།

། ཕྲ་རགས་རྣམ་པར་དབྱེ་བ་རྒྱུད་ལ་འཆང༌། ། སྲུང་མཚམས་ལས་འདས་དབྱིངས་སུ་གྲོལ་བ་ཡིས། ། གསང་སྔགས་ལམ་གྱི་དེ་ཉིད་མངོན་གྱུར་ཤོག ༡༠ ། བསྐྱེད་རྫོགས་མཛེས་པའི་ཁ་ཕྲུས་བྲན་པ་ལས། ། རྩ་རླུང་ཐིག་ལེའི་གདུང་བ་རབ་བཅིལ་ཏེ། ། སྐུ་བཞིའི་པད་དཀར་རྒྱས་པའི་བསིལ་གཡབ་ཏུ། ། མ་ལུས་འགྲོ་བའི་བསྟི་གནས་འབྱུང་བར་ཤོག ༡༡ ། དོན་དམ་གང་དུའང་མི་གནས་ཡེ་ཤེས་སེམས། ། ལྷན་ཅིག་དགའ་བའི་རྩེ་དགས་རོལ་པར་ཤོག ། ཚུར་མཐོང་རྣམ་པར་གྲོལ་བའི་འཁོར་ལོ་ཆེ། ། སྒྱུ་འཕྲུལ་དྲྭ་བའི་ངོ་བོར་རྟག་བརྟན་ཤོག ༡༢ ། བསླབ་གསུམ་ཁྲིམས་ཀྱི་བདུད་རྩི་རྟག་རོལ་ཞིང༌། ། ཟུང་འཇུག་བདེ་དང་སྟོང་པས་རབ་མྱོས་པའི། ། རིག་གྲོལ་གཉིས་མེད་དགེ་འདུན་འདུས་པའི་སྡེས། ། སྲིད་པའི་ཁམས་ཀུན་དོག་པོར་བགྱིད་པར་ཤོག ༡༣ ། དེ་མཐུས་ཚོགས་གཉིས་དཔྱིད་ཀྱིས་ཉེར་དྲངས་པའི། ། རྫོགས་ལྡན་དབྱར་གྱི་འབྱོར་པར་སྤྱོད་པ་ཡིས། ། འགྲོ་ཀུན་ཉོན་མོངས་ཚོགས་རྣམས་རབ་ཞི་སྟེ། ། ཆོས་མཐུན་དགེ་བའི་དཔལ་གྱིས་འཚོ་བར་ཤོག ༡༤ ། འདིས་མཚོན་རྣམ་དག་སྨོན་པའི་དྲི་བཞོན་གྱིས། ། བསླབ་གསུམ་སྤྲིན་གྱི་བླ་བྲེས་བསྐུལ་བ་ལས། ། མངོན་མཐོ་ངེས་ལེགས་བདུད་རྩིའི་དཔལ་འཇོ་བས། ། འགྲོ་ཀུན་སྲིད་པའི་ཚ་གདུང་བཅིལ་བར་ཤོག ༡༥ །ཅེས་པའང་སྲིད་ལས་ངེས་པར་གྲོལ་བའི་བློས་བསླབ་པ་རིན་པོ་ཆེ་དམ་དུ་འཆང་བ་ནི་བསྟན་ལ་སྨོན་པ་ཡོངས་ཀྱི་ངེས་བརྗོད་དུ་སྣང་བའི་ཕྱིར། དེའི་རྗེས་སུ་སྨོན་པ་ཙམ་དུ་རྨོངས་བརྟུལ་ཨ་བྷཱ་ཡའི་མིང་གིས་སོ།། །།
