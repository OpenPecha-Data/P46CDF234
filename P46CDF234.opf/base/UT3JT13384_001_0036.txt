༄༅། །རྗེ་བཙུན་བློ་བཟང་གྲགས་པའི་ལྟ་གྲུབ་ཀྱི་བཞེད་ཚུལ་མདོ་ཙམ་བཞུགས་སོ།།ན་མོ་མཉྫུ་གྷོ་ཥཱ་ཡ། འཇམ་དཔལ་དབྱངས་ཀྱིས་གསང་བའི་མཛོད་ཆེན་པོ་བྱིན་ཅིང༌། ཤིང་རྟ་ཆེན་པོའི་གཞུང་ལུགས་ལ་བརྟེན་ནས་མདོ་སྔགས་ཐམས་ཅད་གདམས་པར་ཤར་བ་རྗེ་བཙུན་བློ་བཟང་གྲགས་པའི་དཔལ་གྱིས་ལྟ་གྲུབ་ཇི་ལྟར་བཞེད་ཚུལ་རྒྱས་པ་ནི། བྱང་ཆུབ་ལམ་རིམ་ཆེན་མོ་དང༌། རྩ་འཇུག་གི་རྣམ་བཤད། དྲང་ངེས་རྣམ་འབྱེད། སྲས་ཀྱི་ཐུ་བོ་གཉིས་ཀྱིས་མཛད་པའི་སྟོང་ཐུན་སྐལ་བཟང་མིག་འབྱེད་དང༌། ལྟ་ཁྲིད་ལྟ་ངན་མུན་སེལ། ཁྱད་པར་སྔགས་ཀྱི་བདེ་སྟོང་སྦྱོར་ཚུལ་གསང་བ་འདུས་པའི་རིམ་ལྔ་གསལ་སྒྲོན་སོགས་ལས་ཤེས་པར་བྱ་བ་ལས་སྨོས་ཀྱིས་མི་ལངས་ཀྱང༌། མདོར་བསྡུ་ན། འཁོར་འདས་ཀྱི་ཆོས་ཐམས་ཅད་རང་རང་གི་གདགས་གཞིའི་ངོས་ནས་དོན་དམ་པར་གྲུབ་པ་རྡུལ་ཙམ་ཡང་མི་དམིགས་ཀྱང་གདགས་གཞིའི་སྟེང་དུ་མིང་དང་རྟོག་པས་ཕར་བཞག་ཙམ་དེ་ཐ་སྙད་དུ་ཚད་མས་གྲུབ་ལ། ཆོས་ཐམས་ཅད་རང་རང་གི་ངོ་བོས་སྟོང་པའི་རྒྱུ་མཚན་གྱིས་ཐ་སྙད་དུ་དེ་དང་དེར་སྣང་ཞིང་བྱ་བྱེད་སོགས་འཐད་ལྡན་དུ་འཇོག་ཤེས་པས། རང་བཞིན་གྱིས་གྲུབ་པ་ཐ་སྙད་དུ་ཡང་ཁེགས་ཀྱང༌། ལས་འབྲས་ལ་སོགས་པའི་བྱ་བྱེད་ཐམས་ཅད་གཞན་ངོ་ཙམ་ལ་སྐྱེལ་མི་དགོས་པར་རང་ལུགས་ལ་ཡང་ཚད་གྲུབ་ཏུ་ཁས་བླངས་པས་ཆོག་པར་མ་ཟད། རྟེན་འབྲེལ་གྱི་ཆོས་ལ་ངེས་པ་དྲངས་པ་ཉིད་ཀྱིས་རང་བཞིན་མེད་པ་ལ་ངེས་པ་ཆེར་འགྲོ་བས་སྟོང་དང་རྟེན་འབྱུང་གཅིག་གྲོགས་གཅིག་འཁྱེར་ཤེས་པ་ཞིག་ཡོང་ཞེས་གསུངས། གཞན་དུ་ན་སྔ་རབས་པ་དག་དོན་དམ་དཔྱོད་བྱེད་ཀྱི་རིགས་པས་དཔྱད་ནས་དགག་བྱ་མ་ལུས་པ་ཁེགས་པ་དང༌། བཀག་པའི་ཤུལ་དུ་ལས་འབྲས་བྱ་བྱེད་སོགས་བསྙོན་མེད་དུ་འཇོག་ཤེས་པའི་གཉིས་ཚོགས་དེ་སུས་ཀྱང་མ་རྟོགས་པས་རྣམ་པར་འཁྱམས་སོ། །དེ་ལྟར་རྟོགས་ནས་ཉམས་སུ་བླངས་པའི་ཚེ་དང༌། འཇོག་སྒོམ་ཁོ་ན་མ་ཡིན་པར་སྐབས་སུ་གང་བབས་ཀྱིས་སྤེལ་བ་དང༌།

ཁྱད་པར་ལས་དང་པོ་པའི་དུས་སུ་ངའོ་སྙམ་པའི་འཛིན་སྟང་དམ་ཧྲིང་བ་དེ་ཡུལ་དུ་བྱས་ནས་དེ་འགོགས་བྱེད་ཀྱི་རིག་པས་བཀག་པའི་གསལ་བའི་ངར་དང་མ་བྲལ་བ་ཅིས་ཀྱང་གལ་ཆེ་སྟེ། གཞན་དུ་ན་འབེན་མ་མཐོང་བའི་མདའ་བཞིན་དུ་དོན་མེད་པར་འགྱུར་བའི་ཕྱིར་རོ། འདི་ལྟར་གྱི་ལྟ་སྒོམ་གྱི་གནད་རྣམ་པར་དག་པ་རྙེད་ཅིང༌། བསོད་ནམས་ཀྱི་ཚོགས་མཐའ་ཡས་པས་གྲོགས་བྱས་ནས་ཉོན་ཤེས་སྒྲིབ་པ་རགས་པ་སྤང་ནུས་ཀྱང༌། ཕྲ་མོ་སྤང་མི་ནུས་པས་འཚང་རྒྱ་བ་ལ་ཡུན་རིང་དུ་འགོར་བས། དེས་ན་སྙིགས་དུས་ཀྱི་ཚེ་ཐུང་གཅིག་ལ་ཤེས་སྒྲིབ་མ་ལུས་པ་སྤང་ནུས་པའི་རྡོ་རྗེ་ཐེག་པ་བླ་མེད་ཀྱི་ལམ་ལ་འཇུག་ཅིང༌། །དེར་ཞུགས་མཁན་གྱི་གང་ཟག་ཀྱང་ཕར་ཕྱིན་ཐེག་པའི་དབང་པོ་རྣོ་ཤོས་ལས་ཀྱང་ཆེས་རྣོ་བ་ཞིག་དགོས་པས། དེ་འདྲ་བའི་གདུལ་བྱ་དེས་དབུ་མ་ཐལ་འགྱུར་བས་ཇི་ལྟར་བཀྲལ་བའི་སྟོང་པ་ཉིད་དམིགས་རྣམ་དག་རྙེད་པའི་ཚེ། བླ་མེད་ཀྱི་དབང་བསྐུར་བ་ཟབ་མོ་དང༌། བསྐྱེད་པའི་རིམ་པ་སྔོན་དུ་འགྲོ་བས། རྫོགས་རིམ་རླུང་སྦྱོར་དང་གཏུམ་མོ་སོགས་ལ་བརྟེན་ནས་རླུང་དབུ་མར་ཞུགས་གནས་ཐིམ་གསུམ་གྱི་ནུས་པས་ཡིད་བདེ་བའི་ངོ་བོར་སོང་བས་སྔར་གྱི་ཡུལ་སྟོང་པ་ཉིད་དུ་རྟོགས་པ་དེ་གང་ཞིག ཡུལ་བདེ་བ་ཆེན་པོ་དང༌། ཡུལ་ཅན་སྟོང་ཉིད་རྟོགས་པའི་ལྷག་མཐོང་ཟུང་དུ་འཇུག་པ་ནི་རྒྱུད་སྡེ་རྣམས་དང༌། གྲུབ་པའི་སློབ་དཔོན་རྣམས་ཀྱིས་བཞེད་པའི་བདེ་སྟོང་ཟུང་འཇུག་མ་ནོར་བ་ཡིན་པས། ཤིང་རྟ་ཆེན་པོ་ཀླུ་སྒྲུབ་ཀྱིས་ཇི་ལྟར་བཀྲལ་བའི་དོན་དཔལ་ལྡན་ཟླ་བས་གསལ་བར་མཛད་པའི་གཞུང་ལ་མ་བརྟེན་པར་སྔགས་བླ་མེད་ཀྱིས་ཀྱང་གྲུབ་པ་ཐོབ་པར་མི་སྲིད་དེ། དེའི་ཉམས་ལེན་མཐར་ཐུག་པ་ནི་བདེ་སྟོང་ཟུང་འཇུག་ཡིན་ན། གང་གི་ཡ་གྱལ་དུ་བཞག་པའི་སྟོང་པ་ཉིད་དམིགས་རྣམ་དག་རྙེད་ཚུལ་མ་སྟོན་པའི་ཕྱིར་རོ། །དེས་ན་སེམས་ཙམ་དུའང་མ་སོན་པའི་ལྟ་བས་སྔགས་བླ་མེད་ཀྱི་བདེ་བ་དང་སྦྱོར་བའི་སྟོང་ཉིད་དུ་མི་རུང་ཞིང༌།

དེ་མི་རུང་བས་ན་བསྐྱེད་རྫོགས་ཙམ་སྒོམ་ལོ་བྱས་ཀྱང་སྲོག་མེད་པའི་གཟུགས་བརྙན་དུ་སོང་བས་ན་ཆོས་སྐུའི་ལག་རྗེས་འཇོག་རྒྱུ་མེད་པས་གྱི་ན་ཙམ་དུ་ཟད་ཅིང༌། ལྟ་བ་རྣམ་པར་དག་པ་འདིར་བསྟན་པ་བཞིན་རྙེད་ནས་བདེ་བ་ཆེན་པོའི་ཐབས་ཀྱིས་ཟིན་པ་དང༌། ཁྱད་པར་དེའི་སྟེང་ཚེ་འདིར་མཚན་དཔེས་སྤྲས་པའི་རྡོ་རྗེའི་སྐུ་ཐོབ་པར་བྱེད་པ་ནི་སྒྱུ་མའི་སྐུ་སྒྲུབ་ཚུལ་ཤེས་མི་ཤེས་ཁོ་ན་ལ་རག་ལས་པས་དེའི་ངོས་འཛིན་འགོས་ལུགས་སུ་གསལ་བར་མ་གསུངས་ཤིང༌། །མར་ལུགས་པ་དག །ལུས་ལྷ་སྐུ་ལ་བྱས་ནས་དེ་སྣང་ལ་རང་བཞིན་མེད་པར་ངོ་སྤྲོད་བྱེད་པ་ཙམ་དུ་འདོད་ཀྱང༌། འདིར་ནི་སྒྱུ་ལུས་ཀྱི་སྒྲུབ་གཞི་གཏན་ལ་དབབས་ནས་སེམས་དབེན་ལ་གནས་པའི་རྣལ་འབྱོར་པ་དེས་ནང་གི་རྐྱེན་རྡོར་བཟླས་དང༌། ཕྱིའི་རྐྱེན་ཕྱག་རྒྱ་དང་སྙོམས་པར་ཞུགས་པས་ས་ཆུ་ལ་ཐིམ་པ་སོགས་ཀྱི་རིམ་པས་རླུང་སེམས་གཉིས་འཆི་རིམ་ཇི་ལྟ་བ་བཞིན་དུ་ཞུགས་པའི་སེམས་དབེན་སྟོང་པ་གསུམ་དུ་བསྒྱུར་ནས། སེམས་དབེན་མཐར་ཐུག་འོད་གསལ་ལས་ལུགས་ལྡོག་ཏུ་ལྡང་བ་ན། དེའི་བཞོན་པ་གཉུག་མའི་རླུང་དེ་ཡིས་རིགས་འདྲ་ཕྱི་མ་དེ་མ་དག་པའི་སྒྱུ་ལུས་སུ་ལྡང་ཚུལ་གསལ་བར་བཤད་དོ། །དེས་ན་ཐུན་མོངས་པའི་ལམ་ངེས་འབྱུང་བྱང་སེམས། ཡང་དག་པའི་ལྟ་བ་གསུམ་ལ་སྔ་མ་སྔ་མ་ལ་མྱོང་བ་ཐོབ་ནས་ཕྱི་མ་ཕྱི་མ་བསྒོམས་ཤིང༌། ལྟ་བའི་དཔྱད་པ་ཡོངས་སུ་རྫོགས་པ་ན། སྔགས་བླ་མེད་ལ་ཞུགས་ཏེ་བདེ་སྟོང་ཟུང་འཇུག་གི་ལྷན་ཅིག་སྐྱེས་པའི་ཡེ་ཤེས་དང༌། སྒྱུ་ལུས་ཀྱི་གནད་ཁྱད་པར་ཅན་ལ་བརྟེན་ནས་ཚེ་འདི་ཉིད་ལ་ཕྱག་རྒྱ་ཆེན་པོ་མཆོག་གི་དངོས་གྲུབ་ལ་སྦྱོར་བ་ཡིན་ནོ། །ཞེས་གསུངས་སོ། །ཞེས་པའང་རང་ཉིད་ཀྱི་བརྗེད་ཐོར་ཟིན་བྲིས་སུ་བགྱིས་པ་ཤུ་བྷཾ།། །།
