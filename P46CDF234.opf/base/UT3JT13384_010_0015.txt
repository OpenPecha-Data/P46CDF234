༁ྃ༁ྃ༔ ཐུགས་རྗེ་ཆེན་པོ་དོན་གསལ་སྒྲོན་མེ་ལས༔ བརྒྱུད་པའི་གསོལ་འདེབས་བཞུགས༔ 

ཨེ་མ་ཧོ༔ རྣམ་དག་སྤྲོས་བྲལ་ཆོས་སྐུའི་ཕོ་བྲང་ནས༔ རིག་སྟོང་དབྱེར་མེད་འོད་དཔག་མེད་མགོན་གྱིས༔ སྐལ་ལྡན་བུ་ལ་དབང་བསྐུར་བྱིན་གྱིས་རློབས༔ འཁོར་འདས་རང་གྲོལ་རྟོགས་པར་གསོལ་བ་འདེབས༔ རིག་རྩལ་འགག་མེད་འོད་ལྔའི་དཀྱིལ་འཁོར་ནས༔ ལོངས་སྤྱོད་རྫོགས་སྐུ་སྤྱན་རས་གཟིགས་དབང་གིས༔ སྐལ་ལྡན་བུ་ལ་དབང་བསྐུར་བྱིན་གྱིས་རློབས༔ རང་སྣང་བརྟན་པ་ཐོབ་པར་གསོལ་བ་འདེབས༔ སྣ་ཚོགས་འགྲོ་འདུལ་སྤྲུལ་སྐུའི་ཞིང་ཁམས་ནས༔ འདིའི་མ་ཕྱིར་སྤྱན་རས་གཟིགས་ཐམས་ཅད་ལ་སྤྱན་རལ་གཟིགས་ཞེས་བྲིས་སྣང༌། དེའང་གཏེར་གཞུང་རང་གི་ངེས་ཚིག་བཤད་པ་དང་བསྟུན་ན་རྫོགས་ཆེན་གྱི་གཞུང་རྣམས་ལས་ལྷུན་གྲུབ་ཕྱི་རྒྱ་རལ་ཅེས་གསུངས་པ་དང་གོ་བ་གནད་གཅིག །དོན་དུ་ཐུགས་རྗེ་ཡུལ་ལ་འཇུག་པའི་སྤྱན་དབྱེ་བ་ལ་དགོངས་པ་ཡིན་འདྲ་བས་འགལ་བ་མེད་རུང༌། བྲིས་པས་དཔེ་བཤུ་དུས་སྤྱན་རས་གཟིགས་ཞེས་བཅོས་བཞག་པ་རེད་པས་ཕྱི་རབས་རྣམས་ཀྱིས་དེ་ལྟར་དགོངས་མཛོད། མཆན། འགྲོ་བ་འདྲེན་མཛད་བྲམ་ཟེ་རྣམ་གསུམ་གྱིས༔ སྐལ་ལྡན་བུ་ལ༴ ཁམས་གསུམ་འཁོར་བ་སྟོངས་པར༴ རྔ་ཡབ་གླིང་ཕྲན་ཟངས་མདོག་དཔལ་རི་ནས༔ སངས་རྒྱས་ཀུན་འདུས་ཨོ་རྒྱན་པདྨ་ཡིས༔ སྐལ་ལྡན༴ བདག་གཞན་དོན་གཉིས་འགྲུབ་པར༴ ཀློང་ཡངས་རྣམ་དག་རིག་པའི་ཞིང་ཁམས་ནས༔ འབྲེལ་ཚད་དོན་ལྡན་ལས་འཕྲོ་གླིང་པ་ཡིས༔ སྐལ་ལྡན༴ རིག་པའི་རྩལ་དབང་ཐོབ་པར༴ རྩ་ཁམས་རྣམ་དག་ཐིག་ལེའི་དཀྱིལ་འཁོར་ནས༔ ཐུགས་རྗེའི་རང་བཞིན་ཞི་ཁྲོའི་ལྷ་ཚོགས་ཀྱིས༔ སྐལ་ལྡན༴ བསྐྱེད་རྫོགས་མཐའ་རུ་ཕྱིན་པར༴ བལ་ཡུལ་གླང་བྲག་གསང་བའི་གཏེར་གནས་ནས༔ གསང་བའི་ཟབ་ཆོས་དོན་གསལ་སྒྲོན་མེ་ཡིས༔ སྐལ་ལྡན༴ མདོ་སྔགས་བསྟན་པ་རྒྱས་པར༴ བཀའ་དང་དམ་ལ་གནས་པའི་གཞལ་ཡས་ནས༔ ཆོས་སྐྱོང་སྲུང་མ་གཏེར་བདག་དམ་ཅན་གྱིས༔ སྐལ་ལྡན་བུ་ལ་དབང་བསྐུར་བྱིན་གྱིས་རློབས༔ ཕྲིན་ལས་རྣམ་བཞི་འགྲུབ་པར་གསོལ་བ་འདེབས༔ བར་ཆད་གདོན་བགེགས་འཇོམས་པར་གསོལ་བ་འདེབས༔

བསམ་པ་ཡིད་བཞིན་འགྲུབ་པར་བྱིན་གྱིས་རློབས༔ ༡༢༣༤༥༦༧༨༩༠༩༨༧༧༦།
