༄༅། །དམ་པ་རིགས་བརྒྱའི་ལྷ་དབང་གི་བཀོལ་བྱང་བཞུགས་སོ། །ཧཱུྃ། ཆོས་རྣམས་རྣམ་དག་ཡེ་ནས་ལྷུན་གྱིས་གྲུབ། །བྱ་བ་བྱེད་དང་ཡིད་ཆོས་རྣམ་པར་དག །ནམ་མཁའི་མཐའ་བཞིན་ཀུན་ལ་སྙོམས་མཛད་པ། །ཀུན་བཟང་ཡབ་ཡུམ་གཉིས་ཀྱིས་དབང་བསྐུར་རོ། །ཧཱུྃ་ཨ་བྷི་ཥིཉྩ་མི། ཧཱུྃ། ཆོས་སྐུ་ནམ་མཁའ་བཞིན་དུ་དབྱེར་མེད་ཀྱང༌། །གཟུགས་སུ་འཇའ་ཚོན་བཞིན་དུ་སོ་སོར་གསལ། །ཐབས་དང་ཤེས་རབ་མཆོག་ལ་མངའ་བརྙེས་པའི། །རིགས་ལྔ་བདེ་བར་གཤེགས་པས་དབང་བསྐུར་རོ། །ཧཱུྃ་ཨོཾ་སྭཱ་ཨཱྃ་ཧཱ་ཨ་བྷི་ཥིཉྩ་མི། ཧཱུྃ། ངོ་མཚར་ཕོ་བྲང་བདེ་བར་གཤེགས་པའི་ཡུམ། །དུས་གསུམ་རྒྱལ་བ་ཐམས་ཅད་སྐྱེད་མཛད་མ། །རིགས་ཀྱི་ཡུམ་གྱུར་རྣམས་ཀྱིས་དབང་བསྐུར་རོ། །མཱུྃ་ལཱྃ་མཱྃ་པྱཱྃ་ཏཱྃ་ཨ་བྷི་ཥིཉྩ་མི། ཧཱུྃ། རྣམ་ཤེས་དབང་པོ་དག་པ་སྐུ་གསུང་ཐུགས། །བདེ་བར་གཤེགས་པ་རྣམས་ཀྱི་སྲས་མཆོག་སྟེ། །སེམས་དཔའ་ཆེན་པོ་བརྒྱད་ཀྱིས་དབང་བསྐུར་རོ། །མ་ཧཱ་ས་ཏྭ་ཨ་བྷི་ཥིཉྩ་མི། ཧཱུྃ། སྒེག་མོ་ཕྲེང་ཐོགས་རིན་ཆེན་གླུ་གར་ཅན། །ནམ་མཁའ་མཉམ་པའི་མཆོད་པས་རྒྱལ་བ་མཆོད། །བྱང་ཆུབ་ལྷ་མོའི་ཚོགས་ཀྱིས་དབང་བསྐུར་རོ། །བོ་དྷི་དེ་ཝཱི་ཨ་བྷི་ཥིཉྩ་མི། ཧཱུྃ། བདུག་སྤོས་མེ་ཏོག་མར་མེ་དྲི་ཆབ་མ། །རྒྱལ་བ་རྒྱ་མཚོ་རྣམས་ལ་མཆོད་འབུལ་མ། །མཆོད་པའི་ལྷ་མོ་རྣམས་ཀྱིས་དབང་བསྐུར་རོ། །པཱུ་ཛ་དེ་ཝཱི་ཨ་བྷི་ཥིཉྩ་མི། ཧཱུྃ། དྲག་ཤུལ་ཐབས་ཀྱིས་འགྲོ་བ་འདུལ་མཛད་པ། །ཚད་མེད་བཞི་ལྡན་སྒོ་བཞི་ཐུབ་པར་སྲུང༌། །ཏ་ཀྲྀད་སྒོ་བ་བཞི་ཡིས་དབང་བསྐུར་རོ། །ཧཱུྃ་ཧཱུྃ་ཧཱུྃ་ཧཱུྃ་ཨ་བྷི་ཥིཉྩ་མི། ཧཱུྃ། རྟག་ཆད་མཐར་ཕྱིན་གསང་བ་ཡུམ་གྱི་ངང༌། །ལྕགས་ཀྱུ་ཞགས་པ་ལྕགས་སྒྲོག་དྲིལ་འཁྲོལ་མ། །སྒོ་མ་ཆེན་མོ་བཞི་ཡིས་དབང་བསྐུར་རོ། །ཛཿཧཱུྃ་བྃ་ཧོཿཨ་བྷི་ཥིཉྩ་མི། ཧཱུྃ། སྟེང་འོག་ཕྱོགས་བཞི་འཇིག་རྟེན་འདུལ་མཛད་གཙོ། །ཉོན་མོངས་དྲུག་འཇོམས་རྒྱུད་དྲུག་དག་པར་སྦྱོང༌། །བཅོམ་ལྡན་ཐུབ་པ་དྲུག་གིས་དབང་བསྐུར་རོ།

།སརྦ་མུ་ནེ་ཨ་བྷི་ཥིཉྩ་མི། ཧཱུྃ། འཇིག་རྟེན་འདས་པའི་དུས་ན་མངོན་བྱུང་བ། །གདུག་པ་འདུལ་ཕྱིར་དུས་ལས་མི་འདའ་བའི། །དཔལ་ཆེན་ཧེ་རུ་ཀ་རྣམས་ཀྱིས་དབང་བསྐུར་རོ། །སརྦ་ཤྲཱི་ཧེ་རུ་ཀ་ཨ་བྷི་ཥིཉྩ་མི། ཧཱུྃ། ཁྲོ་མོ་དབང་ཕྱུག་སྐུ་ལ་འཁྲིལ་མཛད་ཅིང༌། །མཁའ་འགྲོ་མི་བཟད་འབར་བའི་སྤྲིན་འཕྲོ་མ། གདུག་པ་འདུལ་མཛད་ཡུམ་གྱིས་དབང་བསྐུར་རོ། །སརྦ་ཀྲོ་དྷི་ཤྭ་རཱི་ཨ་བྷི་ཥིཉྩ་མི། ཧཱུྃ། བསྐལ་པ་མེ་ལྟར་འབར་བའི་ཀློང་དཀྱིལ་ནས། །ཡེ་ཤེས་སྐུ་མཆོག་འཇིགས་བྱེད་རྒྱན་གྱིས་བཞུགས། །ཁྲོ་མོ་ཆེན་མོ་བརྒྱད་ཀྱིས་དབང་བསྐུར་རོ། །ཧ་ཨ་བྷི་ཥིཉྩ་མི། ཧཱུྃ། ནམ་མཁའི་ལུས་ཅན་གར་ཡང་ཐོགས་མེད་ཅིང༌། །འདོད་དགུར་སྒྱུར་བའི་གཟུགས་ཅན་འཕྲུལ་མོ་ཆེ། །ཕྲ་མེན་ཆེན་མོ་བརྒྱད་ཀྱིས་དབང་བསྐུར་རོ། །ཧེ་ཨ་བྷི་ཨ་བྷི་ཥིཉྩ་མི། ཧཱུྃ། རྡོ་རྗེ་གདོང་མོ་ཕོ་ཉ་མགྱོགས་མ་བཞི། །ཤར་ལྷོ་ནུབ་བྱང་སྒོ་བཞི་ཐུབ་པར་སྲུང༌། །སྒོ་སྐྱོང་ཁྲོ་མོ་བཞི་ཡིས་དབང་བསྐུར་རོ། །ཛཿཧཱུྃ་བྃ་ཧོཿཨ་བྷི་ཥིཉྩ་མི། ཧཱུྃ། དཔལ་གྱི་འཁོར་ཚོགས་བཛྲ་ཨཱ་རལླི། །མཁའ་ལ་ཤུགས་འགྲོ་མ་ཚོགས་དབང་ཕྱུག་མ། །རྣལ་འབྱོར་ཉི་ཤུ་བརྒྱད་ཀྱིས་དབང་བསྐུར་རོ། བྷྱོ་བྷྱོ་བྷྱོ་བྷྱོ་ཨ་བྷི་ཥིཉྩ་མི། ཤིན་ཏུ་བསྡུ་ན། མི་རྟོག་རྡོ་རྗེ་ཐབས་ཀྱི་ཡེ་ཤེས་ནི། །དམིགས་མེད་ཤེས་རབ་ཡུམ་གྱི་ངང་དུ་རྟོགས། །ཐུགས་རྗེའི་བྱེ་བྲག་སྣ་ཚོགས་ཅིར་ཡང་སྟོན། །ཞི་ཁྲོ་རབ་འབྱམས་རྒྱལ་བས་དབང་བསྐུར་རོ། །ཞེས་དང༌། རྩ་སྔགས་སམ། ཡི་གེ་བརྒྱ་པའི་མཐར། བཛྲི་བྷ་ཝ་ཨ་བྷི་ཥིཉྩ་མི། ཞེས་དབང་བསྐུར་བས། འདི་ལྟར་སྣང་བའི་ཆོས་ཐམས་ཅད་དམ་པ་རིགས་བརྒྱའི་ངོ་བོར་དག་ཅིང༌། དེ་ཡང་རང་སེམས་ལྷག་པ་བདེན་གཉིས་དབྱེར་མེད་གསང་ཆེན་རིགས་གཅིག་ཏུ་རྣམ་པར་དག་པ་ཡིན་ནོ། །ཞེས་རྟོག་པ་བསྐྱེད་དོ།། །།
