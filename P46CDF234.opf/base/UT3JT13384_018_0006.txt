བཀྲ་ཤིས་མཚན་གྱི་དབྱངས་སྙན་ཐོས་པས་ཀྱང༌། ། ཕུན་ཚོགས་སྲིད་ཞིའི་དགེ་ལེགས་སྦྱིན་མཛད་པ། ། རྩ་བརྒྱུད་བླ་མ་དཀོན་མཆོག་རྣམ་གསུམ་སོགས། ། སྐྱབས་གནས་རྒྱ་མཚོའི་ཚོགས་ལ་ཕྱག་འཚལ་ལོ། ། ལུས་དང་ལོངས་སྤྱོད་ཀུན་བཟང་མཆོད་སྤྲིན་འབུལ། ། ཚེ་རབས་བསགས་པའི་སྡིག་ལྟུང་བཤགས་ཤིང་བསྡམ། ། འཕགས་དང་སོ་སྐྱེའི་དགེ་བར་རྗེས་ཡི་རངས། ། ཟབ་རྒྱས་ཆོས་ཀྱི་འཁོར་ལོ་བསྐོར་བཞིན་དུ། ། རྟག་པར་མྱ་ངན་མི་འདའ་བཞུགས་གསོལ་འདེབས། ། དགེ་བ་འདིས་མཚོན་དུས་གསུམ་ལེགས་བྱས་ཚོགས། ། འགྲོ་ཀུན་ཀུན་མཁྱེན་གོ་འཕང་ཐོབ་བྱའི་ཕྱིར། ། བྱང་ཆུབ་སྙིང་པོར་ཡོངས་སུ་བསྔོ་བར་བགྱི། ། བསོད་ནམས་གང་དེས་བདག་གཞན་ཡིད་ཅན་ཀུན། ། དེང་ནས་བཟུང་སྟེ་ཚེ་རབས་ཐམས་ཅད་དུ། ། འཁོར་ལོ་བཞི་དང་ལྡན་པའི་རྟེན་ཐོབ་ནས། ། ཐེག་མཆོག་བཤེས་དང་ནམ་ཡང་མི་འབྲལ་ཤོག ། མཉེས་པ་གསུམ་གྱིས་དམ་པའི་ཞབས་བསྟེན་ནས། ། ཇི་ལྟར་ཁས་བླངས་ཚུལ་ཁྲིམས་རྣམ་པར་དག ། མང་དུ་ཐོས་དང་བསམ་པའི་ཕ་རོལ་ཕྱིན། ། རྩེ་གཅིག་ཟབ་མོའི་དོན་མཆོག་བསྒོམ་པར་ཤོག ། དལ་འབྱོར་རྙེད་དཀའ་འཇིག་སླའི་ངང་ཚུལ་ཅན། ། འཁོར་བའི་སྡུག་བསྔལ་དྲན་པའི་ངེས་འབྱུང་གིས། ། དགེ་སྡིག་བླང་དོར་ཚུལ་བཞིན་ལེགས་སྤྱོད་པའི། ། སྔོན་འགྲོ་རྟེན་གྱི་ཆོས་ལ་བློ་འབྱོངས་ཤོག ། གཟུང་བའི་ཆོས་ཀུན་རྨི་ལམ་ལྟ་བུར་རྟོགས། ། འཛིན་པའི་ཤེས་པ་མ་སྐྱེས་རིག་པའི་གཤིས། ། སྤྲོས་བྲལ་བློ་འདས་དང་དུ་མཉམ་བཞག་པས། ། བཞེན་གཉིས་ཟུང་འཇུག་གནས་ལུགས་མངོན་གྱུར་ཤོག ། བདག་གི་བདེ་དགེ་སེམས་ཅན་རྣམས་ལ་གཏོང༌། ། གཞན་གྱི་སྡུག་བསྔལ་ཀུན་འབྱུང་རང་ལ་བླངས། ། རྣམ་པ་ཀུན་ཏུ་འགྲོ་ལ་ཕན་བརྩོན་པའི། ། རླབས་ཆེན་བྱང་ཆུབ་སེམས་མཆོག་འབྱོར་བར་ཤོག ། རྐྱེན་ངན་ཐམས་ཅད་བྱང་ཆུབ་ལམ་དུ་གྱུར། ། ཚེ་གཅིག་ཉམས་ལེན་སྟོབས་ལྔའི་དམ་བཅའ་བརྟན། ། བློ་སྦྱོང་ཚད་ཕྱིན་བསླབ་བྱ་དམ་ཚིག་རྣམས། ། ནམ་ཡང་ཉམས་མེད་སྲོག་བཞིན་བསྲུང་ནུས་ཤོག
